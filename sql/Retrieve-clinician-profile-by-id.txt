-- 1. Wipe out the old parameters entirely
DROP FUNCTION IF EXISTS get_assigned_clinician( UUID);

CREATE OR REPLACE FUNCTION get_assigned_clinician(p_clinician_id UUID)
RETURNS TABLE (
    name TEXT,
    surname TEXT,
    registration_number TEXT,
    genre TEXT,
    birth_date DATE,
    region TEXT
) 
SECURITY DEFINER -- 🚀 Bypasses RLS roadblocks by using system-level clearance safely
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        c.name, 
        c.surname, 
        c.registration_number, 
        c.genre, 
        c.birth_date,
        c.region
    FROM public."Clinicals" c
    WHERE c.id = p_clinician_id;
END;
$$ LANGUAGE plpgsql;
